.1 العمليات الحسابية Arithmetic Operators

JavaScript بتدعم كل العمليات الحسابية الأساسية، وفيها عمليتان إضافيتان مهمتان:

➕ العمليات الحسابية
const a = 10, b = 3;

a + b   // 13  — جمع
a - b   // 7   — طرح
a * b   // 30  — ضرب
a / b   // 3.333... — قسمة
a % b   // 1   — باقي القسمة (Modulus)
a ** b  // 1000 — أس (10³)
باقي القسمة % — أهم عملية في البرمجة!
🍕 تخيل توزيع بيتزا:

عندك 10 قطع وتريد توزيعهم على 3 أشخاص. كل شخص ياخد 3 قطع (10 ÷ 3 = 3)، يفضل قطعة واحدة زيادة. ده هو 10 % 3 = 1.

💡 استخدامات % في البرمجة
// هل العدد زوجي؟
10 % 2 === 0   // true  ← زوجي
7  % 2 === 0   // false ← فردي

// الدوران على قائمة (مثلاً 0,1,2,0,1,2,...)
index % list.length
زيادة ونقصان سريع:
++ و --
let x = 5;
x++;      // x = 6  (زيادة 1)
x--;      // x = 5  (نقصان 1)
++x;      // x = 6  (زيادة قبل الاستخدام)

.2 عمليات التعيين Assignment Operators

بدل ما تكتب x = x + 5، فيه اختصارات أسهل:

📝 اختصارات التعيين
let x = 10;

x += 5;    // x = x + 5  → 15
x -= 3;    // x = x - 3  → 12
x *= 2;    // x = x * 2  → 24
x /= 4;    // x = x / 4  → 6
x %= 4;    // x = x % 4  → 2
x **= 3;   // x = x ** 3 → 8

.3 عمليات المقارنة Comparison Operators

بترجع دايماً true أو false. هي أساس كل الشروط في البرمجة.

⚖️ المقارنة
5 >  3    // true  — أكبر من
5 <  3    // false — أصغر من
5 >= 5   // true  — أكبر أو يساوي
5 <= 4   // false — أصغر أو يساوي
5 == 5   // true  — يساوي (مع تحويل النوع)
5 === 5  // true  — يساوي (نوع وقيمة)
5 != 3   // true  — لا يساوي
5 !== '5' // true  — لا يساوي (نوع أو قيمة)

.4 العمليات المنطقية Logical Operators

بتربط شرطين أو أكثر معاً:

المعاملالاسمالمعنىمثال
&&AND (و)الاثنان trueage > 18 && hasID
||OR (أو)واحد على الأقل trueisAdmin || isOwner
!NOT (ليس)عكس القيمة!isLoggedIn
🔗 أمثلة
const age     = 20;
const hasCard = true;

// AND: الاثنان لازم true
age >= 18 && hasCard   // true

// OR: واحد كافي
age >= 21 || hasCard   // true (hasCard وحدها كافية)

// NOT: عكس
!hasCard                // false
!(5 > 10)               // true
Short-circuit evaluation — اختصار ذكي:
⚡ Short Circuit
// && يوقف عند أول false
false && doSomething()   // doSomething لن تُنفَّذ أبداً

// || يوقف عند أول true
const name = userName || 'زائر';
// لو userName فارغ، تأخذ 'زائر' كقيمة افتراضية

// ?? — Nullish Coalescing (أحدث)
const val = data ?? 'افتراضي';
// يستخدم الافتراضي فقط لو data = null أو undefined

.5 == مقابل === Equality vs Strict Equality

هذا الفرق من أهم الأشياء في JavaScript — كثير من الأخطاء سببها الخلط بينهما:

⚠️ الفرق المهم
// == يقارن القيمة فقط (يحول الأنواع تلقائياً)
5   ==  '5'   // true  ← خطير!
0   ==  false // true  ← خطير!
''  ==  false // true  ← خطير!
null == undefined // true

// === يقارن القيمة والنوع معاً (آمن)
5   === '5'   // false ← صحيح
0   === false // false ← صحيح
5   === 5     // true  ← صحيح
القاعدة: استخدم === و !== دائماً. تجنب == و != تماماً إلا في حالات محددة جداً.
5 == '5'
true
5 === '5'
false
0 == false
true
0 === false
false

.6 عمليات النصوص String Operations

في JS، عامل + مع النصوص يعني دمج (Concatenation)، مش جمع. وهذا يسبب مفاجآت لو ما انتبهتش:

📝 + مع النصوص
// دمج نصوص
'مرحبا' + ' ' + 'أحمد'   // "مرحبا أحمد"

// رقم + نص = نص!
5 + '3'                      // "53" ← مش 8!
'5' + 3                      // "53"

// باقي العمليات تحول النص لرقم
'10' - 4                     // 6
'10' * 2                     // 20
'10' / 2                     // 5
مشكلة شائعة: لما تاخد قيمة من input المستخدم، بتكون نص دائماً. لو تعمل عليها + هتلاقيها بتعمل دمج مش جمع. الحل: حوّلها لرقم أولاً بـ Number() أو + قبلها.
✅ الحل الصحيح
const input = '5';  // نص من المستخدم

// ❌ خطأ
input + 3           // "53"

// ✅ صح
Number(input) + 3  // 8
+input + 3          // 8 (+ أمام المتغير يحوله لرقم)

.7 أولوية التنفيذ Operator Precedence

زي الرياضيات، JS بتنفّذ العمليات بترتيب معين. الأقواس دائماً لها الأولوية العليا.

🔢 ترتيب التنفيذ
// بدون أقواس — ضرب قبل جمع
2 + 3 * 4       // 14 (3*4=12, ثم 2+12)

// مع أقواس — غيّرنا الترتيب
(2 + 3) * 4    // 20 (2+3=5, ثم 5*4)

// ترتيب الأولوية (من الأعلى للأسفل):
// 1. ()  — أقواس
// 2. **  — أس
// 3. * / % — ضرب وقسمة وباقي
// 4. + -  — جمع وطرح
// 5. > < >= <= == === — مقارنة
// 6. && — AND
// 7. || — OR
📌 مثال مركّب: 
const result = 2 + 3 * 4 > 10 && 5 !== '5';
// الخطوات:
// 1. 3 * 4 = 12
// 2. 2 + 12 = 14
// 3. 14 > 10 = true
// 4. 5 !== '5' = true
// 5. true && true = true
نصيحة: لو مش متأكد من الأولوية — استخدم أقواس دايماً. الكود يصبح أوضح وتتجنب الأخطاء.

📝 التمارين العملية Exercises

تمرين 1: العمليات الحسابية

احسب: مجموع 17 و 8، ناتج ضرب 6 في 7، وباقي قسمة 25 على 4. اطبع النتائج.

الحل: 
console.log(17 + 8);   // 25
console.log(6 * 7);    // 42
console.log(25 % 4);   // 1
تمرين 2: هل العدد زوجي؟

استخدم عامل % لمعرفة إذا كان العدد 42 زوجياً أم فردياً. النتيجة true أو false.

الحل: 
const num = 42;
const isEven = num % 2 === 0;
console.log(isEven);  // true
تمرين 3: اختصارات التعيين

عندك متغير let points = 100. أضف 50، ثم اطرح 30، ثم اضرب في 2. اطبع النتيجة في كل خطوة.

الحل: 
let points = 100;
points += 50;  console.log(points);  // 150
points -= 30;  console.log(points);  // 120
points *= 2;   console.log(points);  // 240
تمرين 4: مقارنة صارمة

قارن بين 10 و '10' بـ == و ===. ثم قارن 5 و 5 بنفس الطريقتين. اشرح الفرق.

الحل: 
console.log(10 ==  '10');  // true  (تحوّل النص لرقم)
console.log(10 === '10');  // false (نوع مختلف)
console.log(5  ==  5);     // true
console.log(5  === 5);     // true
تمرين 5: منطقي مركّب

عندك: age = 17 و hasPermission = true. اكتب شرطاً يتحقق إذا كان العمر 18+ أو لديه إذن. ثم اكتب شرطاً يتحقق إذا كان العمر بين 13 و 17.

الحل: 
const age           = 17;
const hasPermission = true;

const canEnter = age >= 18 || hasPermission;
console.log(canEnter);  // true

const isTeen = age >= 13 && age <= 17;
console.log(isTeen);    // true
