.5 منطق JavaScript JS Logic

1 إضافة مهمة
➕ addTodo
function addTodo() {
  const text = input.value.trim();
  if (!text) return;

  const todo = {
    id: Date.now(),
    text,
    completed: false
  };

  todos.unshift(todo); // إضافة للأول
  input.value = '';
  saveTodos();
  renderTodos();
}
2 عرض المهام
🖥️ renderTodos
function renderTodos() {
  const filtered = todos.filter(t => {
    if (currentFilter === 'done')   return t.completed;
    if (currentFilter === 'active') return !t.completed;
    return true;
  });

  list.innerHTML = filtered.map(t => `
    <li class="todo-item ${t.completed ? 'done' : ''}"
        data-id="${t.id}">
      <input type="checkbox" ${t.completed ? 'checked' : ''}
             onchange="toggleTodo(${t.id})">
      <span class="todo-text">${t.text}</span>
      <button class="del-btn" onclick="deleteTodo(${t.id})">🗑️</button>
    </li>
  `).join('');

  updateStats();
}

.6 Local Storage Persistence

💾 حفظ وتحميل البيانات
const KEY = 'my-todos';

function saveTodos() {
  localStorage.setItem(KEY, JSON.stringify(todos));
}

function loadTodos() {
  const saved = localStorage.getItem(KEY);
  return saved ? JSON.parse(saved) : [];
}

// عند بدء التطبيق:
let todos = loadTodos();
renderTodos();

.7 الإحصائيات والفلاتر Stats & Filters

📊 updateStats و toggleFilter
function updateStats() {
  const done  = todos.filter(t => t.completed).length;
  const total = todos.length;
  statTotal.textContent = `${total} مهمة`;
  statDone.textContent  = `${done} مكتملة`;
}

// ربط أزرار الفلتر:
document.querySelectorAll('.filter-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.filter-btn')
      .forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    currentFilter = btn.dataset.filter;
    renderTodos();
  });
});

تحدي 1: تحرير المهام

أضف زر "تعديل" لكل مهمة يتيح تغيير نصها عند الضغط عليه.

function editTodo(id) {
  const todo = todos.find(t => t.id === id);
  const newText = prompt("عدّل المهمة:", todo.text);
  if (newText?.trim()) {
    todo.text = newText.trim();
    save(); render();
  }
}
تحدي 2: أولوية المهام

أضف حقل اختيار أولوية (عالية / متوسطة / منخفضة) مع لون مختلف لكل أولوية.

الفكرة:

أضف حقل select في نموذج الإضافة، وخزّن priority في كل مهمة. ثم أضف CSS class بناءً على الأولوية.

تحدي 3: تاريخ الاستحقاق

أضف حقل تاريخ لكل مهمة، واعرض تحذيراً إذا اقترب الموعد النهائي.

الفكرة: 
const daysLeft = (dueDate) => {
  const diff = new Date(dueDate) - new Date();
  return Math.ceil(diff / (1000*60*60*24));
};
// إذا daysLeft <= 1 أظهر تحذير أحمر
