.2 حلقة for for Loop

الأكثر استخداماً — مناسبة لما تعرف عدد التكرارات مسبقاً:

🔄 شكل for
for (let i = 0; i < 5; i++) {
  console.log(i);
}
// 0, 1, 2, 3, 4

// الأجزاء الثلاثة:
// let i = 0    → نقطة البداية
// i < 5        → شرط الاستمرار
// i++          → التحديث بعد كل دورة
أمثلة شائعة:
📋 أمثلة for
// العد من 1 إلى 10
for (let i = 1; i <= 10; i++) {
  console.log(i);
}

// العد التنازلي
for (let i = 10; i >= 1; i--) {
  console.log(i);
}

// تجاوز كل عنصر في مصفوفة بالـ index
const fruits = ['تفاح', 'موز', 'برتقال'];
for (let i = 0; i < fruits.length; i++) {
  console.log(`${i}: ${fruits[i]}`);
}
for (let i=1; i<=3; i++) console.log(i)

.3 حلقة while while Loop

مناسبة لما لا تعرف عدد التكرارات مسبقاً — تستمر طالما الشرط true:

🔁 while
let count = 1;

while (count <= 5) {
  console.log(count);
  count++;   // مهم جداً! بدونه حلقة لا نهائية
}
// 1, 2, 3, 4, 5
📌 مثال عملي: البحث عن عدد 
let num = 1;

// ابحث عن أول عدد قابل للقسمة على 7 و 3 معاً
while (num % 7 !== 0 || num % 3 !== 0) {
  num++;
}
console.log(num);  // 21
خطر الحلقة اللانهائية! لو نسيت تحديث المتغير أو كتبت شرطاً لا يصبح false أبداً، المتصفح سيتجمد. دائماً تأكد من وجود خروج من الحلقة.

.4 حلقة do...while do...while Loop

مثل while لكن تنفّذ مرة واحدة على الأقل قبل فحص الشرط:

🔂 do...while
let x = 10;

do {
  console.log('نُفِّذ!');   // ينفّذ مرة واحدة
  x++;
} while (x < 5);  // الشرط false من البداية لكن...

// الفرق: while يفحص أولاً، do...while ينفّذ أولاً
الحلقةمتى تستخدم؟
forعدد التكرارات معروف مسبقاً
whileتستمر حتى يتحقق شرط معين
do...whileتنفيذ مرة أولى مضمون (مثل: اطلب إدخال)

.5 for...of — للمصفوفات for...of

أنظف طريقة للتجوال على عناصر مصفوفة — بدون index:

🎯 for...of
const colors = ['أحمر', 'أخضر', 'أزرق'];

for (const color of colors) {
  console.log(color);
}
// أحمر
// أخضر
// أزرق
📝 مقارنة: for vs for...of
const nums = [10, 20, 30];

// for عادي (عندك الـ index)
for (let i = 0; i < nums.length; i++) {
  console.log(nums[i]);
}

// for...of (أنظف، بدون index)
for (const n of nums) {
  console.log(n);
}
forEach أيضاً: طريقة ثالثة شهيرة — nums.forEach(n => console.log(n)) — سنتعلمها في درس المصفوفات.

.6 for...in — للكائنات for...in

للتجوال على خصائص (keys) كائن (Object):

🔑 for...in
const person = {
  name: 'أحمد',
  age:  25,
  city: 'القاهرة'
};

for (const key in person) {
  console.log(`${key}: ${person[key]}`);
}
// name: أحمد
// age: 25
// city: القاهرة
for...of مع الكائنات: لا يعمل مباشرة. استخدم Object.keys() أو Object.values() أو Object.entries() أولاً.

.7 break و continue Loop Control

أداتان للتحكم في سير الحلقة:

⏹️ break — إيقاف الحلقة
// أوقف الحلقة لما تجد 5
for (let i = 0; i < 10; i++) {
  if (i === 5) break;
  console.log(i);
}
// 0, 1, 2, 3, 4
⏭️ continue — تخطي دورة
// تخطي الأعداد الفردية
for (let i = 0; i < 10; i++) {
  if (i % 2 !== 0) continue;
  console.log(i);
}
// 0, 2, 4, 6, 8
الكلمةالوظيفةمثال على الاستخدام
breakيوقف الحلقة نهائياًوجدت ما تبحث عنه
continueيتخطى الدورة الحالية ويكملتجاهل قيم معينة

📝 التمارين العملية Exercises

تمرين 1: جدول الضرب

اطبع جدول ضرب الرقم 5 من 1 إلى 10 بهذا الشكل: "5 × 1 = 5".

الحل: 
for (let i = 1; i <= 10; i++) {
  console.log(`5 × ${i} = ${5 * i}`);
}
تمرين 2: مجموع الأعداد الزوجية

احسب مجموع كل الأعداد الزوجية من 1 إلى 100.

الحل: 
let sum = 0;
for (let i = 1; i <= 100; i++) {
  if (i % 2 === 0) sum += i;
}
console.log(sum);  // 2550
تمرين 3: عكس مصفوفة

عندك مصفوفة [1, 2, 3, 4, 5]. استخدم for لطباعة عناصرها بالعكس.

الحل: 
const arr = [1, 2, 3, 4, 5];
for (let i = arr.length - 1; i >= 0; i--) {
  console.log(arr[i]);
}
// 5, 4, 3, 2, 1
تمرين 4: البحث في مصفوفة

ابحث عن الرقم 7 في المصفوفة [3, 1, 8, 7, 2, 5]. لو وجدته اطبع index موقعه وأوقف البحث.

الحل: 
const nums = [3, 1, 8, 7, 2, 5];
for (let i = 0; i < nums.length; i++) {
  if (nums[i] === 7) {
    console.log(`وجدت 7 في index ${i}`);
    break;
  }
}
تمرين 5: طباعة خصائص كائن

عندك كائن { brand: 'Toyota', year: 2020, color: 'أبيض' }. استخدم for...in لطباعة كل خاصية وقيمتها.

الحل: 
const car = { brand: 'Toyota', year: 2020, color: 'أبيض' };

for (const key in car) {
  console.log(`${key}: ${car[key]}`);
}
