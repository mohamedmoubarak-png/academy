.5 جلب بيانات الطقس Fetching Data

1 دالة جلب الطقس
🌐 getWeather
async function getWeather(city) {
  const KEY = 'YOUR_API_KEY';
  const url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${KEY}&units=metric&lang=ar`;

  showLoading();

  try {
    const res  = await fetch(url);

    // المدينة غير موجودة
    if (!res.ok) throw new Error('مدينة غير موجودة');

    const data = await res.json();
    displayWeather(data);

  } catch (err) {
    showError(err.message);
  }
}
2 عرض بيانات الطقس
🖥️ displayWeather
function displayWeather(data) {
  const { name, main, weather, wind, sys } = data;

  card.innerHTML = `
    <div class="city-name">
      ${name}, ${sys.country}
    </div>
    <div class="weather-icon">
      <img src="https://openweathermap.org/img/wn/${weather[0].icon}@2x.png">
    </div>
    <div class="temp">${Math.round(main.temp)}°C</div>
    <div class="desc">${weather[0].description}</div>
    <div class="details">
      💧 ${main.humidity}%
      💨 ${wind.speed} م/ث
      🌡️ يبدو ${Math.round(main.feels_like)}°
    </div>
  `;
  showCard();
}

.6 معالجة الأخطاء والتحميل Error Handling

⚠️ حالات التطبيق
const loading = document.getElementById('loading');
const card    = document.getElementById('weatherCard');
const errBox  = document.getElementById('errorMsg');

function showLoading() {
  loading.classList.remove('hidden');
  card.classList.add('hidden');
  errBox.classList.add('hidden');
}

function showCard() {
  loading.classList.add('hidden');
  card.classList.remove('hidden');
}

function showError(msg) {
  loading.classList.add('hidden');
  errBox.classList.remove('hidden');
  errBox.textContent = `❌ ${msg}`;
}
أخطاء شائعة في Fetch:
• 401 — API Key خاطئ أو غير نشط بعد
• 404 — اسم المدينة غير موجود
• Network Error — لا يوجد اتصال بالإنترنت

.7 التوقعات المستقبلية 5-Day Forecast

📅 جلب توقعات 5 أيام
async function getForecast(city) {
  const url = `https://api.openweathermap.org/data/2.5/forecast?q=${city}&appid=${KEY}&units=metric&lang=ar&cnt=5`;
  const res  = await fetch(url);
  const data = await res.json();

  // كل 3 ساعات — خذ قراءة يومية واحدة
  const daily = data.list.filter((_, i) => i % 8 === 0);

  forecastEl.innerHTML = daily.map(d => `
    <div class="forecast-day">
      <div>${new Date(d.dt_txt).toLocaleDateString('ar',{weekday:'short'})}</div>
      <img src="https://openweathermap.org/img/wn/${d.weather[0].icon}.png">
      <div>${Math.round(d.main.temp)}°</div>
    </div>
  `).join('');
}

📝 تحديات إضافية Challenges

تحدي 1: API حقيقي

احصل على مفتاح مجاني من openweathermap.org واستبدل البيانات المحاكاة ببيانات حقيقية.

الفكرة: 
const KEY = 'your_real_api_key';
const url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${KEY}&units=metric&lang=ar`;
const res  = await fetch(url);
const data = await res.json();
تحدي 2: تبديل الوحدات

أضف زراً للتبديل بين المئوية (°C) والفهرنهايت (°F).

الفكرة: 
// تحويل من مئوي لفهرنهايت
const toF = (c) => (c * 9/5 + 32).toFixed(1);
const toC = (f) => ((f - 32) * 5/9).toFixed(1);
تحدي 3: الموقع الجغرافي

أضف زر "موقعي الحالي" يستخدم navigator.geolocation لعرض طقس المستخدم تلقائياً.

الفكرة: 
navigator.geolocation.getCurrentPosition(pos => {
  const { latitude, longitude } = pos.coords;
  // استخدم API بـ lat/lon بدل اسم المدينة
  const url = `...?lat=${latitude}&lon=${longitude}&...`;
});
تحدي 4: آخر عمليات البحث

احفظ آخر 5 مدن بحث في localStorage وأظهرها كاقتراحات سريعة.

الفكرة: 
const history = JSON.parse(localStorage.getItem('search-history') || '[]');
// بعد كل بحث ناجح:
const updated = [city, ...history.filter(c => c !== city)].slice(0, 5);
localStorage.setItem('search-history', JSON.stringify(updated));
تحدي 5: خلفية ديناميكية

غيّر خلفية التطبيق تلقائياً حسب حالة الطقس (مشمس = صفراء، ممطر = زرقاء رمادية...).

الفكرة: 
const bgMap = {
  'clear':  'linear-gradient(#f97316, #fbbf24)',
  'clouds': 'linear-gradient(#475569, #94a3b8)',
  'rain':   'linear-gradient(#1e3a8a, #475569)',
};
const main = data.weather[0].main.toLowerCase();
document.body.style.background = bgMap[main] || bgMap.clouds;

اختبر فهمك لمشروع Weather App!

1يجب استخدام async/await مع fetch لأن طلبات الشبكة غير متزامنة.
2fetch() يُرجع الـ JSON مباشرة بدون الحاجة لـ .json().
3API Key يجب تخزينه في .env وعدم رفعه لـ GitHub.
4try/catch ضروري مع fetch لمعالجة أخطاء الشبكة والـ API.
5res.ok تكون true دائماً بغض النظر عن كود الحالة HTTP.
