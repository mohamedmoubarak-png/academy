.1 وحدة path — المسارات path module

وحدة path تتعامل مع مسارات الملفات بشكل آمن على كل الأنظمة (Windows يستخدم \ ولينكس /).

path.join()
يدمج أجزاء المسار بأمان
join('src','data','file.json')
→ src/data/file.json
path.resolve()
يُحوّل لمسار مطلق كامل
resolve('src','app.js')
→ /home/user/project/src/app.js
path.basename()
اسم الملف من المسار
basename('/src/app.js')
→ app.js
path.dirname()
اسم المجلد من المسار
dirname('/src/app.js')
→ /src
path.extname()
امتداد الملف
extname('app.js')
→ .js
path.parse()
يُحلّل المسار لأجزائه
parse('/src/app.js')
→ { root, dir, base, ext, name }
📂 path — أمثلة عملية
const path = require('path');

// ✅ الطريقة الصحيحة لبناء المسارات
const dbPath = path.join(__dirname, 'data', 'db.json');
// → /project/data/db.json  (يعمل على Windows وLinux)

// ❌ الطريقة الخاطئة
const wrong = __dirname + '/data/db.json';
// → لا يعمل صح على Windows (يستخدم \)

// تحقق من امتداد الملف
const file = 'image.png';
if (path.extname(file) === '.png') {
  console.log('صورة PNG ✓');
}
استخدم path.join(__dirname, ...) دائماً بدل دمج النصوص — يمنع أخطاء المسارات على بيئات مختلفة.

.2 وحدة fs — الملفات File System

الدالةالوصفمثال
readFileSync()قراءة ملف (تزامني)fs.readFileSync('f.txt', 'utf8')
readFile()قراءة ملف (غير تزامني)fs.readFile('f.txt', 'utf8', cb)
writeFileSync()كتابة ملف (يُنشئ أو يُعيد الكتابة)fs.writeFileSync('f.txt', data)
writeFile()كتابة ملف (غير تزامني)fs.writeFile('f.txt', data, cb)
appendFileSync()إضافة للنهاية (لا يمسح المحتوى)fs.appendFileSync('log.txt', msg)
existsSync()تحقق من وجود ملف/مجلدfs.existsSync('./data')
mkdirSync()إنشاء مجلدfs.mkdirSync('uploads', {recursive:true})
readdirSync()قائمة محتويات مجلدfs.readdirSync('./src')
unlinkSync()حذف ملفfs.unlinkSync('temp.txt')
statSync()معلومات الملف (حجم، تاريخ)fs.statSync('f.txt').size
📄 fs — أمثلة عملية
const fs   = require('fs');
const path = require('path');

// إنشاء مجلد إذا لم يكن موجوداً
const dir = path.join(__dirname, 'uploads');
if (!fs.existsSync(dir)) {
  fs.mkdirSync(dir, { recursive: true });
  console.log('✓ مجلد uploads أُنشئ');
}

// كتابة JSON في ملف
const users = [{ id: 1, name: 'أحمد' }];
fs.writeFileSync(
  path.join(dir, 'users.json'),
  JSON.stringify(users, null, 2),
  'utf8'
);

// قراءة الملف وتحويله
const data = fs.readFileSync(
  path.join(dir, 'users.json'), 'utf8'
);
console.log(JSON.parse(data)); // [{id:1, name:'أحمد'}]

.3 Blocking vs Non-Blocking المشكلة

Node.js به خيط واحد — إذا انتظر قراءة ملف أو استعلام قاعدة بيانات، فالطلبات الأخرى تتوقف. هذه هي مشكلة Blocking.

🔴 Blocking (تزامني)
→ طلب قراءة الملف
⏸ يوقف كل شيء...
⏸ ينتظر الملف...
→ يرد على الطلب
⚠️ الطلبات الأخرى تنتظر!
🟢 Non-Blocking (غير تزامني)
→ طلب قراءة الملف
→ يكمل الطلبات الأخرى ✓
→ يكمل الطلبات الأخرى ✓
✓ الملف جاهز — يُعالجه
→ كل الطلبات تخدمت!
readFileSync مناسبة للإعداد عند بدء التشغيل فقط. في داخل الـ API، استخدم دائماً النسخة الـ Async.

.4 Callbacks — الطريقة الأولى Callbacks

الـ Callback دالة تُمرَّر لدالة أخرى وتُنفَّذ عندما تنتهي المهمة. كانت الطريقة الوحيدة في Node.js القديم.

📞 Callbacks — الأساس
const fs = require('fs');

// النمط: (err, data) — خطأ أولاً دائماً
fs.readFile('data.txt', 'utf8', (err, data) => {
  if (err) {
    console.error('خطأ:', err.message);
    return; ← مهم: أوقف التنفيذ
  }
  console.log('المحتوى:', data);
});

console.log('هذا يظهر قبل قراءة الملف!');
// ← لأن fs.readFile لا تُوقف التنفيذ
😱 Callback Hell — لماذا نتجنبه
// قراءة → معالجة → كتابة → إرسال بريد
fs.readFile('input.txt', 'utf8', (err, data) => {
  if (err) return handleErr(err);
  processData(data, (err, result) => {
    if (err) return handleErr(err);
    fs.writeFile('out.txt', result, (err) => {
      if (err) return handleErr(err);
      sendEmail(result, (err) => {        ← الهرم الأبدي!
        if (err) return handleErr(err);
        console.log('تم!'); ← 4 مستويات عمق
      });
    });
  });
});

.5 Promises — الطريقة الأفضل Promises

الـ Promise وعد بنتيجة مستقبلية — إما نجاح (resolve) أو فشل (reject).

🤝 Promises — المفهوم والاستخدام
// إنشاء Promise
function readUserFromDB(id) {
  return new Promise((resolve, reject) => {
    const user = findUser(id);
    if (user) resolve(user);    ← نجاح
    else    reject(new Error('مستخدم غير موجود'));
  });
}

// الاستخدام — .then().catch()
readUserFromDB(42)
  .then(user  => console.log('المستخدم:', user))
  .catch(err  => console.error('خطأ:', err.message))
  .finally(()  => console.log('انتهى (دائماً)'));

// الحل لـ Callback Hell — promise chaining
readFile('input.txt')
  .then(data   => processData(data))
  .then(result => writeFile('out.txt', result))
  .then(()     => sendEmail())
  .catch(err   => handleErr(err)); ← خطأ واحد يكفي
⚡ Promise.all — تشغيل متوازٍ
// تنتظر انتهاء الكل قبل المتابعة
const [users, posts] = await Promise.all([
  fetchUsers(),
  fetchPosts(),
]);
// كلاهما يعملان بالتوازي — أسرع من التسلسل!

.6 Async/Await — الطريقة الحديثة Async / Await

async/await سكّر نحوي فوق الـ Promises — يجعل الكود الغير تزامني يبدو مثل التزامني تماماً.

قديم ❌
Callbacks
Callback Hell
صعب القراءة
أفضل ✓
Promises
.then() .catch()
أنظف لكن verbose
الأحدث ⭐
Async/Await
أوضح وأنظف
يُستخدم دائماً
⚡ Async/Await — القواعد
// 1. الدالة يجب أن تكون async
async function getUser(id) {

  // 2. await يُوقف التنفيذ حتى ينتهي Promise
  const user = await fetchFromDB(id);

  // 3. للأخطاء: try/catch بدل .catch()
  try {
    const posts = await fetchPostsByUser(user.id);
    return { user, posts };
  } catch (err) {
    console.error('فشل جلب البيانات:', err.message);
    throw err; ← أعد رمي الخطأ
  }
}

// ← async دالة تُعيد Promise دائماً
getUser(1).then(d => console.log(d));
خطأ شائع: نسيان await أمام الـ Promise — الكود يُكمل بدون انتظار النتيجة! الـ TypeScript يُنبّهك، في JS العادية إحذر.

.7 fs.promises — التطبيق الكامل Async File Operations

وحدة fs تحتوي fs.promises التي تُعيد Promises — استخدمها مع async/await.

📁 fs.promises مع async/await
const fs   = require('fs').promises; ← أو: const {promises: fs}
const path = require('path');

async function processUserFile() {
  const filePath = path.join(__dirname, 'users.json');

  try {
    // قراءة الملف
    const raw   = await fs.readFile(filePath, 'utf8');
    const users = JSON.parse(raw);

    // معالجة البيانات
    const active = users.filter(u => u.active);

    // كتابة ملف جديد
    const outPath = path.join(__dirname, 'active-users.json');
    await fs.writeFile(outPath, JSON.stringify(active, null, 2));

    // قائمة الملفات في المجلد
    const files = await fs.readdir(__dirname);
    console.log('الملفات:', files);

    return active;

  } catch (err) {
    if (err.code === 'ENOENT') {
      console.error('الملف غير موجود!');
    } else {
      throw err;
    }
  }
}

processUserFile();
أكواد خطأ fs الشائعة: ENOENT = الملف غير موجود | EACCES = لا صلاحية | EEXIST = موجود مسبقاً

.8 قراءة ملف — حي Live Demo

سكريبت كامل يقرأ ملف JSON ويكتب ملفاً جديداً، مع معالجة الأخطاء:

📂 fs + async/await — مثال كامل Node.js 
// process-data.js
const fs = require('fs').promises;
const path = require('path');

async function run() {
  const src = path.join(
    __dirname, 'data.json'
  );

  // تحقق من وجود الملف
  const exists = await fs
    .access(src)
    .then(() => true)
    .catch(() => false);

  if (!exists) {
    console.log('إنشاء ملف البيانات...');
    await fs.writeFile(src,
      JSON.stringify(
        [{ id:1, name:'أحمد' }],
        null, 2
      )
    );
  }

  // قراءة ومعالجة
  const raw = await fs.readFile(
    src, 'utf8'
  );
  const users = JSON.parse(raw);
  console.log('المستخدمون:', users);
}

run();

📝 التمارين العملية

تمرين 1: path

استخدم path.join لبناء مسار لملف config.json داخل مجلد config في مجلد المشروع الحالي، ثم اطبع:
اسم الملف فقط، المجلد فقط، الامتداد فقط.

الحل: 
const path = require('path');
const filePath = path.join(__dirname, 'config', 'config.json');

console.log(path.basename(filePath)); // config.json
console.log(path.dirname(filePath));  // .../config
console.log(path.extname(filePath));  // .json
تمرين 2: كتابة وقراءة

اكتب سكريبت يُنشئ ملف notes.txt ويكتب فيه 3 ملاحظات (كل واحدة في سطر). ثم يقرأه ويطبع عدد الأسطر.

الحل: 
const fs   = require('fs').promises;
const path = require('path');

async function run() {
  const file = path.join(__dirname, 'notes.txt');
  const content = 'ملاحظة أولى\nملاحظة ثانية\nملاحظة ثالثة';

  await fs.writeFile(file, content, 'utf8');
  const data  = await fs.readFile(file, 'utf8');
  const lines = data.split('\n');
  console.log('عدد الملاحظات:', lines.length); // 3
}
run();
تمرين 3: Async/Await

حوّل هذا الكود من Callbacks لـ async/await:
fs.readFile('a.txt', 'utf8', (err, data) => { if (err) throw err; fs.writeFile('b.txt', data.toUpperCase(), err => { if (err) throw err; console.log('تم'); }); });

الحل: 
const fs = require('fs').promises;

async function convert() {
  try {
    const data = await fs.readFile('a.txt', 'utf8');
    await fs.writeFile('b.txt', data.toUpperCase());
    console.log('تم');
  } catch (err) {
    console.error(err.message);
  }
}
convert();
تمرين 4: Promise.all

اكتب دالة async تقرأ ملفين (users.json و products.json) في آنٍ واحد بـ Promise.all، وتطبع حجم كل مصفوفة.

الحل: 
const fs = require('fs').promises;

async function loadData() {
  const [usersRaw, productsRaw] = await Promise.all([
    fs.readFile('users.json',    'utf8'),
    fs.readFile('products.json', 'utf8'),
  ]);

  const users    = JSON.parse(usersRaw);
  const products = JSON.parse(productsRaw);

  console.log('المستخدمون:', users.length);
  console.log('المنتجات:', products.length);
}
loadData();
تمرين 5: Logger بسيط

اكتب وحدة logger.js تُصدّر دالة log(message) تُضيف كل رسالة في ملف app.log مع التوقيت. استخدمها في app.js.

الحل: 
// logger.js
const fs   = require('fs').promises;
const path = require('path');
const LOG  = path.join(__dirname, 'app.log');

async function log(message) {
  const line = `[${new Date().toISOString()}] ${message}\n`;
  await fs.appendFile(LOG, line, 'utf8');
}

module.exports = { log };

// app.js
const { log } = require('./logger');
async function main() {
  await log('بدأ التطبيق');
  await log('تم معالجة الطلب');
}
main();
