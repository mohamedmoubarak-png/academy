.1 Route Parameters — الـ :param Route Params

الـ Route Parameter هو جزء متغيّر من المسار — تُعرّفه بعلامة : وتجده في req.params:

http://localhost:3000 /api/users/ 42 /posts/ 7 ? page=2&sort=date
Domain
Path ثابت
:param متغيّر
Query String
🔖 Route Parameters الأساسية
// :id في تعريف المسار ← param ديناميكي
app.get('/api/users/:id', (req, res) => {
  console.log(req.params);    // { id: '42' }
  console.log(req.params.id); // '42'  ← دائماً string!

  const id = Number(req.params.id); // حوّله لرقم
  res.json({ userId: id });
});

// اختبار:
// GET /api/users/42   → { userId: 42 }
// GET /api/users/abc  → { userId: NaN } ← تحتاج تحقق!
/api/users/:id ← pattern req.params = { id: '42' }
/api/posts/:slug ← pattern req.params = { slug: 'my-post' }
/files/:filename.:ext ← مع امتداد req.params = { filename: 'logo', ext: 'png' }
⚠️ تنبيه: قيم req.params دائماً strings — حتى لو أرسلت رقماً. تحويلها إلزامي قبل المقارنة الرقمية.

.2 معاملات متعددة في المسار Multiple Params

يمكنك وضع أكثر من param في نفس المسار — كل منها يُخزَّن بمفتاحه في req.params:

🗂️ مسارات متداخلة (Nested Resources)
// مستخدم → منشوراته
app.get('/api/users/:userId/posts', (req, res) => {
  const { userId } = req.params;
  res.json({ userId, posts: [] });
  // GET /api/users/5/posts → userId = '5'
});

// مستخدم → منشور محدد
app.get('/api/users/:userId/posts/:postId', (req, res) => {
  const { userId, postId } = req.params;
  res.json({ userId, postId });
  // GET /api/users/5/posts/12 → { userId:'5', postId:'12' }
});

// مسار عام لـ resource أي يقبل أي نوع
app.get('/api/:resource/:id', (req, res) => {
  const { resource, id } = req.params;
  res.json({ resource, id });
  // GET /api/users/3  → { resource:'users', id:'3' }
  // GET /api/posts/10 → { resource:'posts', id:'10' }
});
المسارات الثابتة أولاً! إذا كان لديك /api/users/me و/api/users/:id، عرّف الثابت أولاً — وإلا سيلتقط :id قيمة "me" ويعاملها كـ ID.
⚠️ الترتيب مهم — ثابت قبل ديناميكي
// ✅ الصحيح — الثابت أولاً
app.get('/api/users/me',  getMeProfile);    // يتطابق مع /me
app.get('/api/users/:id', getUserById);     // يتطابق مع أي شيء آخر

// ❌ الخطأ — :id يلتقط "me"
app.get('/api/users/:id', getUserById);     // ← يمسك كل شيء!
app.get('/api/users/me',  getMeProfile);    // لن يُنفَّذ أبداً

.3 Query String — استعلام الـ URL Query String

الـ Query String هو كل ما يأتي بعد ? في الـ URL — يُستخدم للفلترة والترقيم والبحث. تجده في req.query:

/api/products?page=2&limit=10
req.query.page = "2"
req.query.limit = "10"
/api/search?q=Express&sort=date
req.query.q = "Express"
req.query.sort = "date"
/api/users?role=admin&active=true
req.query.role = "admin"
req.query.active = "true"
/api/products?tags=js&tags=node
req.query.tags = ["js", "node"]
← يصبح array!
🔍 Query String مع قيم افتراضية
app.get('/api/products', (req, res) => {
  // استخراج القيم مع defaults
  const {
    page     = '1',
    limit    = '10',
    sort     = 'id',
    order    = 'asc',
    search   = '',
    category,          // اختياري — قد يكون undefined
  } = req.query;

  // تحويل النصوص لأرقام
  const pageNum  = Math.max(1, Number(page));
  const limitNum = Math.min(100, Math.max(1, Number(limit)));

  res.json({
    page:     pageNum,
    limit:    limitNum,
    sort,
    order,
    search,
    category: category || null,
    total:    0,
    data:     [],
  });
});

// اختبار:
// GET /api/products?page=2&limit=5&sort=price&order=desc
// → { page: 2, limit: 5, sort: 'price', order: 'desc', ... }
القيم دائماً strings: حتى ?page=2 يُعطي "2" وليس 2. استخدم Number() أو parseInt() للتحويل. كذلك ?active=true يُعطي string "true" وليس boolean!

.4 التحقق من الـ Params والـ Query Validation

دائماً تحقق من بيانات الـ URL قبل استخدامها — المستخدم قد يُرسل قيماً خاطئة:

✅ التحقق الكامل من الـ :id
app.get('/api/users/:id', (req, res) => {
  const id = Number(req.params.id);

  // التحقق: رقم صحيح موجب
  if (!Number.isInteger(id) || id <= 0) {
    return res.status(400).json({
      error: 'الـ ID يجب أن يكون رقماً صحيحاً موجباً',
    });
  }

  // البحث في البيانات
  const user = users.find(u => u.id === id);
  if (!user) {
    return res.status(404).json({
      error: `المستخدم رقم ${id} غير موجود`,
    });
  }

  res.json({ data: user });
});
⚙️ Middleware للتحقق من الـ Pagination
// middleware يُعدّ بيانات الترقيم ويتحقق منها
function validatePagination(req, res, next) {
  const page  = Number(req.query.page  || 1);
  const limit = Number(req.query.limit || 10);

  if (isNaN(page) || page < 1) {
    return res.status(400).json({ error: 'page يجب أن يكون >= 1' });
  }
  if (isNaN(limit) || limit < 1 || limit > 100) {
    return res.status(400).json({ error: 'limit بين 1 و 100' });
  }

  // أضف القيم المُعالجة للـ req
  req.pagination = {
    page,
    limit,
    skip: (page - 1) * limit,  // لقاعدة البيانات
  };

  next();  // تابع للـ route handler
}

// استخدام الـ middleware على route محدد
app.get('/api/products', validatePagination, (req, res) => {
  const { page, limit, skip } = req.pagination;
  res.json({ page, limit, skip, data: [] });
});
أين تضع التحقق؟ إذا كان التحقق لـ route واحد، ضعه داخله. إذا تكرر في أكثر من route، حوّله لـ middleware كما في المثال — وهذا ما سنتعمق فيه في درس الـ Middleware.

.5 router.route() — تجميع المسارات Route Chaining

router.route() يُمكّنك من تعريف جميع الـ HTTP methods لنفس المسار في سلسلة واحدة — يُقلّل التكرار ويُحسّن القراءة:

🔗 قبل router.route() — متكرر
// ❌ نكرر '/api/users' في كل سطر
app.get('/api/users',     (req, res) => res.json({ action: 'getAll' }));
app.post('/api/users',    (req, res) => res.json({ action: 'create' }));

app.get('/api/users/:id',    (req, res) => res.json({ action: 'getOne' }));
app.put('/api/users/:id',    (req, res) => res.json({ action: 'replace' }));
app.patch('/api/users/:id',  (req, res) => res.json({ action: 'update' }));
app.delete('/api/users/:id', (req, res) => res.json({ action: 'delete' }));
✅ مع router.route() — أنيق
const router = require('express').Router();

// المسار الجذر /api/users
router.route('/')
  .get((req, res)  => res.json({ action: 'getAll' }))
  .post((req, res) => res.json({ action: 'create' }));

// المسار مع ID /api/users/:id
router.route('/:id')
  .get((req, res)    => res.json({ action: 'getOne',   id: req.params.id }))
  .put((req, res)    => res.json({ action: 'replace',  id: req.params.id }))
  .patch((req, res)  => res.json({ action: 'update',   id: req.params.id }))
  .delete((req, res) => res.json({ action: 'delete',   id: req.params.id }));

module.exports = router;
🏗️ مع handler functions منفصلة
// عادةً تُعرَّف الدوال في controller منفصل
const { getAll, create, getOne, update, remove } = require('./controllers/users');

router.route('/')
  .get(getAll)
  .post(create);

router.route('/:id')
  .get(getOne)
  .patch(update)
  .delete(remove);

.6 Router منفصل لكل Resource Modular Router

الهيكل الاحترافي: ملف Router مستقل لكل نوع من البيانات (users, posts, products...):

📁 routes/users.js — ملف Router مستقل
const router = require('express').Router();

const users = [
  { id: 1, name: 'أحمد', email: 'ahmed@mail.com' },
  { id: 2, name: 'سارة', email: 'sara@mail.com' },
];

// GET / → /api/users
router.get('/', (req, res) => {
  res.json({ data: users, count: users.length });
});

// GET /:id → /api/users/:id
router.get('/:id', (req, res) => {
  const user = users.find(u => u.id === Number(req.params.id));
  if (!user) return res.status(404).json({ error: 'غير موجود' });
  res.json({ data: user });
});

// POST / → /api/users
router.post('/', (req, res) => {
  const { name, email } = req.body;
  if (!name || !email) return res.status(400).json({ error: 'ناقص' });
  const user = { id: Date.now(), name, email };
  users.push(user);
  res.status(201).json({ data: user });
});

// DELETE /:id → /api/users/:id
router.delete('/:id', (req, res) => {
  const idx = users.findIndex(u => u.id === Number(req.params.id));
  if (idx === -1) return res.status(404).json({ error: 'غير موجود' });
  users.splice(idx, 1);
  res.sendStatus(204);
});

module.exports = router;
⚙️ app.js — تجميع الـ Routers
const express = require('express');
const app = express();

app.use(express.json());

// تحميل الـ Routers
app.use('/api/users',    require('./routes/users'));
app.use('/api/posts',    require('./routes/posts'));
app.use('/api/products', require('./routes/products'));

// 404 handler — آخر شيء دائماً
app.use((req, res) => {
  res.status(404).json({ error: 'Not Found', path: req.url });
});

app.listen(3000, () => console.log('✅ يعمل :3000'));
لماذا نفصل الـ Routes؟ مشروع بـ 10 entities (users, products, orders...) بدون تنظيم = ملف app.js بـ 500 سطر. بالـ Router المنفصل كل ملف 30-50 سطر — سهل القراءة والاختبار والصيانة.

.7 أنماط متقدمة — Wildcard وOptional Advanced Patterns

Express يدعم أنماط أكثر مرونة في تعريف المسارات:

⭐ Wildcard — المسار العام *
// يلتقط أي شيء يبدأ بـ /api/
app.get('/api/*', (req, res) => {
  res.status(404).json({
    error: 'API endpoint غير موجود',
    path: req.url,
    hint: 'تحقق من الـ docs',
  });
});

// يلتقط كل شيء — 404 handler عام
app.use('*', (req, res) => {
  res.status(404).json({ error: 'الصفحة غير موجودة' });
});
❓ Optional Param — اختياري بـ ?
// :id اختياري — بدونه يجلب الكل
app.get('/api/users/:id?', (req, res) => {
  if (req.params.id) {
    const user = users.find(u => u.id === Number(req.params.id));
    return res.json({ mode: 'single', data: user });
  }
  res.json({ mode: 'all', data: users });
});

// GET /api/users    → { mode: 'all', data: [...] }
// GET /api/users/5  → { mode: 'single', data: {...} }
📋 ملخص أنماط الـ Routes
// ثابت — يتطابق فقط مع /api/users/me
app.get('/api/users/me', ...);

// ديناميكي — :id يتطابق مع أي قيمة
app.get('/api/users/:id', ...);

// متعدد — userId و postId
app.get('/api/users/:userId/posts/:postId', ...);

// اختياري — :id موجود أو لا
app.get('/api/users/:id?', ...);

// Wildcard — أي شيء بعد /files/
app.get('/files/*', ...);

// Regular Expression — ينتهي بـ .json
app.get(/\.json$/, (req, res) => res.json({ format: 'json' }));
ترتيب التعريف = ترتيب التطابق: Express يتحقق من الـ routes بالترتيب من الأعلى للأسفل ويتوقف عند أول تطابق. الثوابت أولاً، ثم الديناميكية، ثم الـ Wildcard، ثم الـ 404 handler دائماً آخراً.

.8 Routing كامل في العمل Full Routing Demo

خادم Express كامل مع Routing منظّم ويشمل params, query, router.route():

🗺️ Routing — users API كاملة routes/users.js 
// routes/users.js
const router =
  require('express').Router();

let users = [
  { id:1, name:'أحمد', role:'admin' },
  { id:2, name:'سارة', role:'user'  },
];

// GET / — مع query فلترة
router.get('/', (req, res) => {
  const { role, page='1' } = req.query;
  let data = [...users];
  if(role) data = data.filter(
    u => u.role === role
  );
  res.json({
    data,
    page: Number(page),
    total: data.length,
  });
});

// route() — تجميع /:id
router.route('/:id')
  .get((req, res) => {
    const u = users.find(
      x => x.id===Number(req.params.id)
    );
    if(!u) return res
      .status(404)
      .json({error:'غير موجود'});
    res.json({ data: u });
  })
  .patch((req, res) => {
    const u = users.find(
      x => x.id===Number(req.params.id)
    );
    if(!u) return res
      .status(404)
      .json({error:'غير موجود'});
    Object.assign(u, req.body);
    res.json({ data: u });
  })
  .delete((req, res) => {
    const i = users.findIndex(
      x => x.id===Number(req.params.id)
    );
    if(i===-1) return res
      .status(404)
      .json({error:'غير موجود'});
    users.splice(i,1);
    res.sendStatus(204);
  });

module.exports = router;

📝 التمارين العملية Exercises

تمرين 1: Route مع التحقق من ID

أنشئ GET /api/products/:id يتحقق أن الـ ID رقم صحيح موجب، ثم يبحث في مصفوفة وهمية. يُعيد 400 إذا كان الـ ID غير صالح، و404 إذا لم يوجد المنتج.

الحل: 
const products = [{ id: 1, name: 'لابتوب' }, { id: 2, name: 'جوال' }];

app.get('/api/products/:id', (req, res) => {
  const id = Number(req.params.id);

  if (!Number.isInteger(id) || id <= 0) {
    return res.status(400).json({ error: 'ID غير صالح' });
  }

  const product = products.find(p => p.id === id);
  if (!product) {
    return res.status(404).json({ error: `المنتج ${id} غير موجود` });
  }

  res.json({ data: product });
});
تمرين 2: مسار متداخل

أنشئ GET /api/users/:userId/orders يُعيد orders لمستخدم محدد. إذا لم يوجد المستخدم أعد 404. البيانات وهمية.

الحل: 
const users  = [{ id: 1, name: 'أحمد' }, { id: 2, name: 'سارة' }];
const orders = [
  { id: 101, userId: 1, total: 250 },
  { id: 102, userId: 1, total: 180 },
  { id: 103, userId: 2, total: 320 },
];

app.get('/api/users/:userId/orders', (req, res) => {
  const userId = Number(req.params.userId);
  const user   = users.find(u => u.id === userId);

  if (!user) return res.status(404).json({ error: 'المستخدم غير موجود' });

  const userOrders = orders.filter(o => o.userId === userId);
  res.json({ user, orders: userOrders, count: userOrders.length });
});
تمرين 3: Search مع Query Validation

أنشئ GET /api/search يقرأ: ?q=&page=&limit=
التحقق: q مطلوب، page >= 1، limit بين 1 و 50.

الحل: 
app.get('/api/search', (req, res) => {
  const { q, page = '1', limit = '10' } = req.query;

  if (!q || q.trim() === '') {
    return res.status(400).json({ error: 'q مطلوب' });
  }

  const pageNum  = Number(page);
  const limitNum = Number(limit);

  if (isNaN(pageNum) || pageNum < 1)
    return res.status(400).json({ error: 'page >= 1' });
  if (isNaN(limitNum) || limitNum < 1 || limitNum > 50)
    return res.status(400).json({ error: 'limit بين 1 و50' });

  res.json({ q: q.trim(), page: pageNum, limit: limitNum, results: [] });
});
تمرين 4: router.route() لـ Posts

استخدم router.route() لبناء CRUD كامل لـ /api/posts:
GET الكل، POST إنشاء، GET واحد بـ :id، PUT تحديث، DELETE حذف.

الحل: 
const router = require('express').Router();
let posts = [{ id: 1, title: 'أول منشور' }];

router.route('/')
  .get((req, res) => res.json({ data: posts }))
  .post((req, res) => {
    const { title } = req.body;
    if (!title) return res.status(400).json({ error: 'العنوان مطلوب' });
    const post = { id: Date.now(), title };
    posts.push(post);
    res.status(201).json({ data: post });
  });

router.route('/:id')
  .get((req, res) => {
    const post = posts.find(p => p.id === Number(req.params.id));
    if (!post) return res.status(404).json({ error: 'غير موجود' });
    res.json({ data: post });
  })
  .put((req, res) => {
    const i = posts.findIndex(p => p.id === Number(req.params.id));
    if (i === -1) return res.status(404).json({ error: 'غير موجود' });
    posts[i] = { ...posts[i], ...req.body };
    res.json({ data: posts[i] });
  })
  .delete((req, res) => {
    const i = posts.findIndex(p => p.id === Number(req.params.id));
    if (i === -1) return res.status(404).json({ error: 'غير موجود' });
    posts.splice(i, 1);
    res.sendStatus(204);
  });

module.exports = router;
تمرين 5: ترتيب الـ Routes

رتّب هذه الـ routes بالترتيب الصحيح واشرح لماذا:
أ) app.get('/api/users/:id', ...)
ب) app.use((req,res) => res.status(404)...)
ج) app.get('/api/users/me', ...)
د) app.get('/api/users', ...)

الحل:

الترتيب الصحيح: د ← ج ← أ ← ب
• د أولاً: GET /api/users (ثابت، الأكثر تحديداً للمجموعة)
• ج ثانياً: GET /api/users/me (ثابت — يجب قبل :id وإلا التقطه)
• أ ثالثاً: GET /api/users/:id (ديناميكي — يلتقط كل ما لم يُعرَّف)
• ب آخراً: 404 handler (يمسك كل ما لم يتطابق)

اختبر فهمك لـ Routing في Express!

1قيم req.params دائماً من نوع string حتى لو أرسلت رقماً في الـ URL.
2يمكن وضع أكثر من param في نفس المسار مثل /users/:userId/posts/:postId.
3req.query يحتوي فقط على أول query parameter في الـ URL.
4ترتيب تعريف الـ routes في Express لا يؤثر على أيها يتطابق مع الطلب.
5يجب تعريف route ثابت مثل /users/me قبل الديناميكي /users/:id.
