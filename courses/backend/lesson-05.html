.1 الـ HTTP Module في Node.js Built-in HTTP

Node.js تأتي مع وحدة http مدمجة — لا تحتاج npm install لأي شيء. بهذه الوحدة تستطيع بناء خادم ويب كامل من الصفر وتفهم كيف تعمل الأطر (Frameworks) مثل Express من الداخل.

🔩
لماذا نتعلم HTTP الخام قبل Express؟

مثل تعلم قيادة السيارة يدوياً قبل الأوتوماتيكي — ستفهم كل ما يحدث خلف الكواليس، وستعرف لماذا نحتاج Express، وستتمكن من إصلاح المشكلات بعمق.

📦 استيراد وحدة http
// CommonJS — الطريقة الافتراضية في Node.js
const http = require('http');

// لا تحتاج npm install — مدمجة 100%
// مثلها: fs, path, crypto, os, events...
الوحدات المدمجة في Node.js: إلى جانب http، لديك fs (ملفات)، path (مسارات)، crypto (تشفير)، os (نظام التشغيل)، events (أحداث) — كلها جاهزة فوراً.

.2 http.createServer() — إنشاء الخادم Create Server

دالة http.createServer() تُنشئ خادماً جديداً. تأخذ callback يُستدعى تلقائياً مع كل طلب HTTP يصل للخادم:

🖥️ أول خادم HTTP
const http = require('http');

// الـ callback يُستدعى مع كل طلب واصل
const server = http.createServer((req, res) => {
  // req = IncomingMessage   (ماذا أراد العميل؟)
  // res = ServerResponse    (ماذا نرد؟)

  res.writeHead(200, { 'Content-Type': 'text/plain; charset=utf-8' });
  res.end('مرحباً بالعالم! 🌍');
});

// ابدأ الاستماع على port 3000
server.listen(3000, () => {
  console.log('✅ الخادم يعمل على http://localhost:3000');
});
▶️ تشغيل الخادم
node server.js
# → ✅ الخادم يعمل على http://localhost:3000

# اختبر في المتصفح أو بـ curl:
curl http://localhost:3000
# → مرحباً بالعالم! 🌍
Port 3000 في التطوير: الـ Port هو "رقم الباب". المتصفح يستخدم 80 (HTTP) أو 443 (HTTPS) تلقائياً. في التطوير نستخدم 3000 أو 4000 أو 8000 لأنها متاحة بدون صلاحيات مدير النظام.

.3 كائن الطلب req Request Object

كائن req يحتوي على كل معلومات الطلب القادم من العميل:

📥 req — الطلب
req.method GET, POST, PUT, DELETE…
req.url المسار الكامل: /api/users?page=2
req.headers object بكل الـ headers المُرسلة
حدث data chunks من الـ body القادمة
حدث end انتهاء إرسال الـ body
📤 res — الاستجابة
res.writeHead() status code + headers معاً
res.setHeader() ضبط header واحد
res.statusCode تعيين كود الحالة مباشرة
res.write() إرسال جزء من الـ body
res.end() إنهاء الاستجابة (مطلوب دائماً)
🔍 فحص معلومات الطلب
const server = http.createServer((req, res) => {
  console.log('Method :', req.method);
  // → GET

  console.log('URL    :', req.url);
  // → /api/users?page=2&sort=name

  console.log('Host   :', req.headers['host']);
  // → localhost:3000

  console.log('Accept :', req.headers['accept']);
  // → application/json

  res.end('OK');
});
تحليل الـ URL: req.url يُعيد المسار كاملاً مع الـ query string. لتقسيمه: new URL(req.url, 'http://localhost') تُعطيك .pathname و .searchParams بشكل نظيف.

.4 كائن الاستجابة res Response Object

كائن res هو أداتك للرد على الطلب. الترتيب مهم جداً — Headers قبل Body دائماً:

📤 الترتيب الصحيح للاستجابة
// الخطوة 1 — حدّد الكود والـ Headers (أولاً)
res.writeHead(200, {
  'Content-Type': 'application/json',
  'Access-Control-Allow-Origin': '*'
});

// الخطوة 2 — اكتب البيانات (اختياري)
res.write('{"users":[');
res.write('{"id":1}');
res.write(']}');

// الخطوة 3 — أنهِ الاستجابة (مطلوب!)
res.end();

// ─── الأكثر شيوعاً: كل شيء في end() ───
res.writeHead(200, { 'Content-Type': 'application/json' });
res.end(JSON.stringify({ status: 'ok' }));
⚠️ قاعدة لا تُكسر: استدعِ res.end() في كل مسار بدون استثناء. إذا نسيته، سيظل الطلب معلقاً — المتصفح ينتظر حتى timeout والخادم يستهلك موارد للاتصال الميت.
🚦 أمثلة متنوعة على الاستجابات
// نص عادي
res.writeHead(200, { 'Content-Type': 'text/plain; charset=utf-8' });
res.end('مرحباً!');

// HTML
res.writeHead(200, { 'Content-Type': 'text/html; charset=utf-8' });
res.end('<h1 dir="rtl">مرحباً!</h1>');

// JSON 200
res.writeHead(200, { 'Content-Type': 'application/json' });
res.end(JSON.stringify({ success: true, data: [] }));

// خطأ 404
res.writeHead(404, { 'Content-Type': 'application/json' });
res.end(JSON.stringify({ error: 'غير موجود' }));

.5 Routing يدوي بـ if/else Manual Routing

بدون Framework، كل الـ routing يصير يدوياً — نتحقق من req.method و req.url في كل طلب:

GET / → الصفحة الرئيسية
GET /api/users → قائمة المستخدمين JSON
POST /api/users → إنشاء مستخدم 201
404 أي مسار آخر → رسالة خطأ JSON
🗺️ Routing يدوي كامل
const server = http.createServer((req, res) => {
  const { method, url } = req;

  // ─── الصفحة الرئيسية ───
  if (method === 'GET' && url === '/') {
    res.writeHead(200, { 'Content-Type': 'text/plain; charset=utf-8' });
    return res.end('مرحباً! الخادم يعمل 🟢');
  }

  // ─── قائمة المستخدمين ───
  if (method === 'GET' && url === '/api/users') {
    const users = [{ id: 1, name: 'أحمد' }, { id: 2, name: 'سارة' }];
    res.writeHead(200, { 'Content-Type': 'application/json' });
    return res.end(JSON.stringify({ data: users, count: users.length }));
  }

  // ─── إنشاء مستخدم ───
  if (method === 'POST' && url === '/api/users') {
    // سنقرأ الـ body في القسم القادم...
    res.writeHead(201, { 'Content-Type': 'application/json' });
    return res.end(JSON.stringify({ message: 'تم إنشاء المستخدم' }));
  }

  // ─── 404 ─── (يُنفَّذ إذا لم يتطابق أي if)
  res.writeHead(404, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({ error: `المسار ${url} غير موجود` }));
});
لماذا return res.end() وليس res.end() فقط؟ الـ return يوقف تنفيذ بقية الدالة. بدونه ستنفذ جميع الـ if blocks وصولاً للـ 404، وتحاول الرد مرتين — مما يُسبب خطأ "Cannot set headers after they are sent".

.6 إعادة JSON JSON Responses

إعادة JSON هو القلب النابض لأي API. تحتاج دائماً ضبط Content-Type الصحيح وتحويل الكائن بـ JSON.stringify():

📦 دوال مساعدة — Helper Functions
// دالة موحّدة لإعادة JSON ← تُجنّب التكرار
function sendJSON(res, statusCode, data) {
  res.writeHead(statusCode, {
    'Content-Type': 'application/json',
    'Access-Control-Allow-Origin': '*',
  });
  res.end(JSON.stringify(data));
}

// دالة خاصة للأخطاء
function sendError(res, statusCode, message) {
  sendJSON(res, statusCode, {
    success: false,
    error: message,
    code: statusCode,
  });
}

// الاستخدام:
sendJSON(res, 200, { data: users, count: 2 });
sendJSON(res, 201, { message: 'تم الإنشاء', id: 42 });
sendError(res, 404, 'المستخدم غير موجود');
sendError(res, 400, 'بيانات غير صالحة');
🏗️ الخادم مع Helper Functions
const http = require('http');

function sendJSON(res, code, data) {
  res.writeHead(code, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify(data));
}

const users = [
  { id: 1, name: 'أحمد', email: 'ahmed@mail.com' },
  { id: 2, name: 'سارة', email: 'sara@mail.com' },
];

const server = http.createServer((req, res) => {
  const { method, url } = req;

  if (method === 'GET' && url === '/api/users') {
    return sendJSON(res, 200, { data: users, count: users.length });
  }

  if (method === 'GET' && url === '/api/health') {
    return sendJSON(res, 200, {
      status: 'ok',
      uptime: Math.floor(process.uptime()),
    });
  }

  sendJSON(res, 404, { error: 'Not Found', path: url });
});

server.listen(3000, () => console.log('🟢 Server on port 3000'));

.7 قراءة Body من POST Reading POST Body

الـ body في Node.js يصل كـ stream — يتدفق كـ chunks (أجزاء). نجمعها عبر حدثي data و end:

📥 دالة readBody — تقرأ الـ body كاملاً
// نُغلّف القراءة في Promise لنستخدم await
function readBody(req) {
  return new Promise((resolve, reject) => {
    let body = '';

    req.on('data', chunk => {
      body += chunk.toString();  // اجمع الـ chunks
    });

    req.on('end', () => {
      try {
        resolve(JSON.parse(body));  // حوّل لـ object
      } catch {
        reject(new Error('JSON غير صالح'));
      }
    });

    req.on('error', reject);
  });
}
🏗️ POST route يقرأ البيانات
// لاحظ: async لاستخدام await
const server = http.createServer(async (req, res) => {
  const { method, url } = req;

  if (method === 'POST' && url === '/api/users') {
    try {
      const body = await readBody(req);
      // body = { name: 'أحمد', email: 'ahmed@mail.com' }

      // التحقق من البيانات
      if (!body.name || !body.email) {
        return sendJSON(res, 400, { error: 'الاسم والبريد مطلوبان' });
      }

      // محاكاة حفظ في قاعدة البيانات
      const newUser = { id: Date.now(), ...body, createdAt: new Date() };
      sendJSON(res, 201, { message: 'تم إنشاء المستخدم', data: newUser });

    } catch (err) {
      sendJSON(res, 400, { error: 'JSON غير صالح' });
    }
  }
});

// اختبار بـ curl:
// curl -X POST http://localhost:3000/api/users \
//      -H "Content-Type: application/json" \
//      -d '{"name":"أحمد","email":"ahmed@mail.com"}'
لماذا Stream وليس مباشرة؟ الـ body يمكن أن يكون ضخماً جداً (ملف، صور). الـ stream يُرسله أجزاءً حتى لا يملأ الذاكرة دفعة واحدة — هذا ما يجعل Node.js سريعاً مع الملفات الكبيرة.

.8 الخادم الكامل + مقارنة HTTP vs Express Full Server + Comparison

هذا الخادم الكامل يجمع كل ما تعلمناه — انظر كيف يعمل في الـ terminal:

⚙️ HTTP Server كامل — Node.js خام server.js 
// server.js — HTTP خام بدون Framework
const http = require('http');

function sendJSON(res, code, data) {
  res.writeHead(code, {
    'Content-Type':'application/json'
  });
  res.end(JSON.stringify(data));
}

function readBody(req) {
  return new Promise((ok, fail) => {
    let b = '';
    req.on('data', c => b += c);
    req.on('end', () => {
      try { ok(JSON.parse(b)); }
      catch { fail(new Error()); }
    });
  });
}

const users = [
  { id:1, name:'أحمد' },
  { id:2, name:'سارة' },
];

const server = http.createServer(
  async (req, res) => {
  const { method, url } = req;

  if(method==='GET'&&url==='/')
    return sendJSON(res,200,
      {msg:'الرئيسية'});

  if(method==='GET'&&
     url==='/api/users')
    return sendJSON(res,200,
      {data:users});

  if(method==='POST'&&
     url==='/api/users'){
    const b=await readBody(req);
    users.push({id:Date.now(),...b});
    return sendJSON(res,201,
      {message:'تم',data:b});
  }

  sendJSON(res,404,
    {error:'Not Found'});
});

server.listen(3000,()=>
  console.log('🟢 يعمل :3000'));

📝 التمارين العملية Exercises

تمرين 1: خادم متعدد المسارات

أنشئ خادم HTTP يرد على ثلاثة مسارات:
• GET / → JSON: { page: "home" }
• GET /about → JSON: { page: "about", version: "1.0" }
• أي مسار آخر → 404 + JSON: { error: "Not Found" }

الحل: 
const http = require('http');

const server = http.createServer((req, res) => {
  const { method, url } = req;
  res.setHeader('Content-Type', 'application/json');

  if (method === 'GET' && url === '/') {
    res.writeHead(200);
    return res.end(JSON.stringify({ page: 'home' }));
  }
  if (method === 'GET' && url === '/about') {
    res.writeHead(200);
    return res.end(JSON.stringify({ page: 'about', version: '1.0' }));
  }
  res.writeHead(404);
  res.end(JSON.stringify({ error: 'Not Found' }));
});

server.listen(3000, () => console.log('🟢 يعمل'));

تمرين 3: لماذا res.end() مطلوب؟

اشرح بكلماتك: لماذا يجب استدعاء res.end() في كل مسار؟ وما الذي يحدث للمستخدم إذا نسيناه؟

الإجابة:

res.end() يُخبر Node.js بأن الاستجابة اكتملت ويُغلق الاتصال. بدونه:
• الاتصال يظل مفتوحاً إلى الأبد
• المتصفح ينتظر حتى timeout (30-60 ثانية عادةً)
• يظهر للمستخدم دوّامة تحميل لا تنتهي
• الخادم يستهلك موارد لكل اتصال معلق

تمرين 4: دالة sendError موحّدة

اكتب sendError(res, code, message) ترد بـ JSON:
{ success: false, error: <message>, code: <statusCode> }
ثم استخدمها في مسار الـ 404 وفي حالة خطأ POST.

الحل: 
function sendError(res, code, message) {
  res.writeHead(code, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({ success: false, error: message, code }));
}

// الاستخدام:
sendError(res, 404, 'المسار غير موجود');
sendError(res, 400, 'بيانات غير صالحة');
sendError(res, 500, 'خطأ داخلي في الخادم');
تمرين 5: POST /api/echo

أنشئ POST /api/echo يستقبل أي JSON ويُعيده كما هو مع إضافة:
receivedAt: التاريخ والوقت الحاليين بصيغة ISO.

الحل: 
if (method === 'POST' && url === '/api/echo') {
  try {
    const body = await readBody(req);
    sendJSON(res, 200, {
      ...body,
      receivedAt: new Date().toISOString(),
    });
  } catch {
    sendError(res, 400, 'JSON غير صالح');
  }
}
