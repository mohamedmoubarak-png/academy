.1 ما هو EventEmitter؟ Event-Driven Model

Node.js يعتمد على نموذج Event-Driven — الكود لا يُنفَّذ تسلسلياً بل استجابةً للأحداث.

📻
تشبيه: راديو FM

EventEmitter مثل محطة إذاعية — تبثّ أحداثاً (emit). من يريد الاستماع يضبط المحطة (on). لا علاقة مباشرة بين المُرسِل والمستقبِل.

📻 EventEmitter — البداية
const { EventEmitter } = require('events');

// إنشاء emitter
const emitter = new EventEmitter();

// الاستماع للحدث 'data'
emitter.on('data', (msg) => {
  console.log('📨 استلمت:', msg);
});

// إطلاق الحدث
emitter.emit('data', 'مرحباً!');
// ← يطبع: 📨 استلمت: مرحباً!

// يمكن تمرير أكثر من معامل
emitter.emit('data', 'رسالة', { from: 'server', time: Date.now() });

.2 on / emit / once / off Methods

.on(event, fn)
يستمع للحدث كلما أُطلق (مستمر)
.emit(event, …)
يُطلق الحدث مع معاملات اختيارية
.once(event, fn)
يستمع مرة واحدة فقط ثم يُلغى
.off(event, fn)
يُلغي مستمع محدد
.removeAll()
يُلغي كل المستمعين لحدث
.listeners(ev)
قائمة المستمعين المسجّلين
⚙️ الطرق الكاملة
const { EventEmitter } = require('events');
const bus = new EventEmitter();

// on — يستمع كل مرة
bus.on('login', (user) => {
  console.log(`سجّل دخول: ${user}`);
});

// once — مرة واحدة فقط
bus.once('server:start', () => {
  console.log('✓ الخادم بدأ');
});

// emit — إطلاق
bus.emit('login', 'أحمد');   // → سجّل دخول: أحمد
bus.emit('login', 'سارة');   // → سجّل دخول: سارة
bus.emit('server:start');     // → ✓ الخادم بدأ
bus.emit('server:start');     // ← لا يُطبع (once انتهى)

// error — حدث خاص يجب التعامل معه
bus.on('error', (err) => {
  console.error('خطأ:', err.message);
});
bus.emit('error', new Error('فشل الاتصال'));
حدث error خاص — إذا أطلقته بدون مستمع، يُوقف Node.js تلقائياً! دائماً سجّل مستمعاً له.

.3 توارث EventEmitter Extending EventEmitter

أفضل ممارسة: بدل استخدام EventEmitter مباشرة، أورّثه في كلاساتك لتصميم نظام أحداث نظيف.

🏗️ كلاس Server مخصص
const { EventEmitter } = require('events');

class MyServer extends EventEmitter {
  constructor() {
    super();
    this.connections = 0;
  }

  connect(user) {
    this.connections++;
    this.emit('connect', user);           ← أطلق حدثاً
    if (this.connections > 100) {
      this.emit('overload', this.connections);
    }
  }

  disconnect(user) {
    this.connections--;
    this.emit('disconnect', user);
  }
}

// الاستخدام
const server = new MyServer();

server.on('connect',    user => console.log(`✓ ${user} اتصل`));
server.on('disconnect', user => console.log(`✗ ${user} قطع`));
server.on('overload',   n    => console.warn(`⚠️ حمل زائد: ${n}`));

server.connect('أحمد');
server.connect('سارة');
هذا النمط يستخدمه Node.js داخلياً — الـ http.Server نفسه يُورّث EventEmitter ويُطلق 'request'، 'connection'، 'error'.

.4 الأحداث المدمجة في Node.js Built-in Events

معظم وحدات Node.js تستخدم EventEmitter داخلياً — هذه الأحداث التي ستصادفها:

📡 أحداث http.Server
const http = require('http');
const server = http.createServer();

server.on('request', (req, res) => {   ← طلب جديد
  res.end('مرحباً!');
});

server.on('connection', (socket) => {  ← اتصال TCP جديد
  console.log('اتصال من:', socket.remoteAddress);
});

server.on('error', (err) => {          ← خطأ في الخادم
  if (err.code === 'EADDRINUSE') {
    console.error('المنفذ مستخدم!');
  }
});

server.on('close', () => {            ← الخادم أُغلق
  console.log('الخادم أُوقف');
});

server.listen(3000);
📂 أحداث process
// التقاط أخطاء غير متوقعة — مهم جداً
process.on('uncaughtException', (err) => {
  console.error('خطأ غير متوقع:', err);
  process.exit(1);
});

process.on('unhandledRejection', (reason) => {
  console.error('Promise غير معالج:', reason);
});

// عند إغلاق التطبيق (Ctrl+C)
process.on('SIGINT', () => {
  console.log('\nجارٍ الإغلاق بأمان...');
  // أغلق الـ DB connections هنا
  process.exit(0);
});

.5 ما هي Streams؟ Streams — Why?

بدل تحميل الكل في الذاكرة، Streams تُعالج البيانات قطعة قطعة (chunk by chunk):

📖 Readable Stream — أحداثه
const fs = require('fs');

const readStream = fs.createReadStream('big-file.txt', {
  encoding: 'utf8',
  highWaterMark: 64 * 1024, ← حجم الـ chunk (64KB)
});

// حدث: قطعة جديدة جاهزة
readStream.on('data', (chunk) => {
  console.log(`📦 chunk: ${chunk.length} bytes`);
});

// حدث: انتهى الملف
readStream.on('end', () => {
  console.log('✓ تمت القراءة');
});

// حدث: خطأ
readStream.on('error', (err) => {
  console.error('خطأ:', err.message);
});
✏️ Writable Stream
const writeStream = fs.createWriteStream('output.txt');

// كتابة قطعة
writeStream.write('سطر أول\n');
writeStream.write('سطر ثانٍ\n');

// إنهاء الكتابة
writeStream.end('سطر أخير\n');

// حدث: انتهت الكتابة
writeStream.on('finish', () => {
  console.log('✓ كُتب الملف');
});

// مثال حقيقي: الـ Response في Express هو Writable!
app.get('/', (req, res) => {
  res.write('بيانات...');  ← writable.write()
  res.end();                ← writable.end()
});

.7 Pipe — ربط التدفقات Piping

pipe() يُوصّل مخرجات Stream لمدخلات آخر — مثل أنابيب المياه.

📄 Readable
createReadStream

🔗 pipe() — نسخ ملف
const fs = require('fs');

const src  = fs.createReadStream('input.txt');
const dest = fs.createWriteStream('output.txt');

// ربط بـ pipe — بديل عشرات الأسطر!
src.pipe(dest);

dest.on('finish', () => console.log('✓ نُسخ الملف'));
🗜️ ضغط ملف بـ zlib + pipe
const fs   = require('fs');
const zlib = require('zlib');

// قراءة → ضغط → كتابة (كل الملف بـ سطرين!)
fs.createReadStream('data.json')
  .pipe(zlib.createGzip())                     ← ضغط gzip
  .pipe(fs.createWriteStream('data.json.gz'));  ← حفظ

// مثال حقيقي: خدمة ملفات بـ Streams
app.get('/download', (req, res) => {
  res.setHeader('Content-Type', 'text/plain');
  fs.createReadStream('large-file.txt').pipe(res);
  // ← يُرسل مباشرة للمستخدم قطعة قطعة
});

.8 EventEmitter حي Live Demo

نظام أحداث لمحادثة بسيطة (Chat Room) باستخدام EventEmitter:

⚡ EventEmitter — Chat Room Node.js 
// chat.js
const { EventEmitter } = require('events');

class ChatRoom extends EventEmitter {
  constructor(name) {
    super();
    this.name  = name;
    this.users = [];
  }

  join(user) {
    this.users.push(user);
    this.emit('join', user);
  }

  send(from, msg) {
    this.emit('message', { from, msg });
  }

  leave(user) {
    this.users =
      this.users.filter(u => u !== user);
    this.emit('leave', user);
  }
}

const room = new ChatRoom('عام');

room.on('join',    u    =>
  console.log(`✓ ${u} انضم`));
room.on('message', ({from, msg}) =>
  console.log(`[${from}]: ${msg}`));
room.on('leave',   u    =>
  console.log(`✗ ${u} غادر`));

room.join('أحمد');
room.join('سارة');
room.send('أحمد', 'هلا!');
room.send('سارة', 'أهلاً!');
room.leave('أحمد');

📝 التمارين العملية

تمرين 1: EventEmitter أساسي

أنشئ EventEmitter يستمع لحدث 'order'. كل مرة يُطلق الحدث مع اسم المنتج وسعره، يطبع رسالة "✓ طلب جديد: [المنتج] — [السعر] ريال". أطلق الحدث 3 مرات بمنتجات مختلفة.

الحل: 
const { EventEmitter } = require('events');
const shop = new EventEmitter();

shop.on('order', (product, price) => {
  console.log(`✓ طلب جديد: ${product} — ${price} ريال`);
});

shop.emit('order', 'قهوة', 15);
shop.emit('order', 'تشيز كيك', 25);
shop.emit('order', 'عصير', 12);
تمرين 2: once

أنشئ EventEmitter يستمع لحدث 'init' مرة واحدة فقط بـ once، وحدث 'request' كل مرة بـ on. أطلق كلا الحدثين 3 مرات وراقب الفرق.

الحل: 
const { EventEmitter } = require('events');
const srv = new EventEmitter();

srv.once('init', () => console.log('✓ تهيئة (مرة واحدة)'));
srv.on('request', (n) => console.log(`طلب #${n}`));

for (let i = 1; i <= 3; i++) {
  srv.emit('init');     // يطبع مرة واحدة فقط
  srv.emit('request', i); // يطبع 3 مرات
}
// النتيجة:
// ✓ تهيئة (مرة واحدة)
// طلب #1  طلب #2  طلب #3
تمرين 3: قراءة بـ Stream

أنشئ ملف نصي كبير (يمكن ملؤه بـ fs.writeFileSync بنص مكرر). ثم اقرأه بـ createReadStream وعدّ عدد الـ chunks التي وصلت.

الحل: 
const fs = require('fs');

// إنشاء ملف اختباري
const big = 'سطر بيانات اختبارية كثيرة جداً! '.repeat(5000);
fs.writeFileSync('test.txt', big, 'utf8');

// قراءة بـ Stream
let chunks = 0;
const rs = fs.createReadStream('test.txt', { highWaterMark: 1024 });

rs.on('data', () => chunks++);
rs.on('end',  () => console.log(`وصلنا ${chunks} chunk`));
تمرين 4: نسخ بـ pipe

اكتب سكريبت ينسخ ملفاً من مسار لآخر باستخدام createReadStream + pipe + createWriteStream، ويطبع "✓ تم النسخ" بعد الانتهاء.

الحل: 
const fs = require('fs');

const src  = fs.createReadStream('source.txt');
const dest = fs.createWriteStream('copy.txt');

src.pipe(dest);
dest.on('finish', () => console.log('✓ تم النسخ'));
تمرين 5: كلاس Logger بالأحداث

أنشئ كلاس Logger يُورّث EventEmitter. يحتوي على دالة log(level, message) تُطلق حدث 'log'. سجّل مستمعاً يطبع الرسائل ملوّنة: INFO أزرق، WARN أصفر، ERROR أحمر (استخدم رموز الأحرف مثل ✅⚠️❌).

الحل: 
const { EventEmitter } = require('events');

class Logger extends EventEmitter {
  log(level, message) {
    this.emit('log', { level, message, time: new Date() });
  }
}

const logger = new Logger();

const icons = { INFO: '✅', WARN: '⚠️', ERROR: '❌' };

logger.on('log', ({ level, message, time }) => {
  const icon = icons[level] || '📝';
  console.log(`${icon} [${level}] ${time.toLocaleTimeString()} — ${message}`);
});

logger.log('INFO',  'الخادم بدأ');
logger.log('WARN',  'الذاكرة عالية');
logger.log('ERROR', 'فشل الاتصال بقاعدة البيانات');
