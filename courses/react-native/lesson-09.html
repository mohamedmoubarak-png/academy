.1 الظلال — Shadow و Elevation Shadows

الظل يُعطي عمقاً وارتفاعاً للعناصر — كأنها تطفو فوق الصفحة. في RN الظلال تختلف بين iOS وAndroid.

🍎 iOS — shadow* properties
const styles = StyleSheet.create({
  // ظل ناعم (iOS)
  cardSoft: {
    shadowColor: '#000',
    shadowOffset: { width: 0, height: 4 },
    shadowOpacity: 0.08,
    shadowRadius: 12,
    backgroundColor: 'white',
  },

  // ظل ملون (مميز)
  cardBlue: {
    shadowColor: '#0ea5e9',   // لون الظل نفسه
    shadowOffset: { width: 0, height: 8 },
    shadowOpacity: 0.35,
    shadowRadius: 16,
    backgroundColor: 'white',
  },
});

// 🤖 Android — elevation فقط
cardAndroid: {
  elevation: 8,           // 0-24، كلما زاد = ظل أقوى
  backgroundColor: 'white',
},

// ✅ الحل الشامل (iOS + Android)
card: {
  backgroundColor: 'white',
  ...Platform.select({
    ios: {
      shadowColor: '#000',
      shadowOffset: { width: 0, height: 4 },
      shadowOpacity: 0.1,
      shadowRadius: 8,
    },
    android: { elevation: 5 },
  }),
},
ملاحظة مهمة: على Android، elevation يُلوّن الظل دائماً بالأسود ولا يمكن تغيير لونه إلا من React Native 0.72+ بـ shadowColor على Android.

.3 الـ Gradient بـ LinearGradient Gradients

React Native لا تدعم الـ gradient مباشرة — نستخدم expo-linear-gradient:

أزرق/بنفسجي
أحمر/برتقالي

أخضر/أزرق
🎨 expo-linear-gradient
// التثبيت
npx expo install expo-linear-gradient

import { LinearGradient } from 'expo-linear-gradient';

// gradient أفقي
<LinearGradient
  colors={['#0ea5e9', '#7c3aed']}
  start={{ x: 0, y: 0 }}
  end={{ x: 1, y: 0 }}
  style={styles.gradient}
>
  <Text style={{ color: 'white', fontWeight: 'bold' }}>
    خلفية Gradient 🎨
  </Text>
</LinearGradient>

// gradient قطري
<LinearGradient
  colors={['#f43f5e', '#f97316', '#fbbf24']}
  start={{ x: 0, y: 0 }}
  end={{ x: 1, y: 1 }}
  style={[styles.card, { borderRadius: 16, padding: 20 }]}
>
  <Text style={{ color: 'white' }}>غروب الشمس 🌅</Text>
</LinearGradient>

// زر gradient
<TouchableOpacity onPress={handlePress}>
  <LinearGradient
    colors={['#0ea5e9', '#0284c7']}
    style={styles.btn}
  >
    <Text style={styles.btnText}>إرسال</Text>
  </LinearGradient>
</TouchableOpacity>
start و end: يحددان اتجاه الـ gradient. {x:0, y:0} = الزاوية العليا اليسرى. {x:1, y:0} = أفقي. {x:0, y:1} = عمودي. {x:1, y:1} = قطري.

👁️ طرق التحكم في الشفافية
// 1. opacity على العنصر كله (يؤثر على الأبناء أيضاً)
<View style={{ opacity: 0.5 }}>
  <Text>هذا النص سيكون شفافاً</Text>
</View>

// 2. rgba في اللون (لا يؤثر على الأبناء)
<View style={{ backgroundColor: 'rgba(14, 165, 233, 0.15)' }}>
  <Text>هذا النص واضح</Text>
</View>

// 3. شفافية ديناميكية — لتأثير Disabled
<TouchableOpacity
  style={{ opacity: isDisabled ? 0.4 : 1 }}
  disabled={isDisabled}
>
  <Text>زر</Text>
</TouchableOpacity>

// 4. ألوان بصيغة hex + alpha
backgroundColor: '#0ea5e926'   // آخر رقمين = الشفافية (26 = 15%)
backgroundColor: '#0ea5e980'   // 80 = 50%
opacity vs rgba: استخدم opacity لتأثيرات الـ disabled أو Fade. استخدم rgba لخلفيات شبه شفافة دون التأثير على النص بداخلها.

🔄 Transform في RN
// في RN، transform = array من objects
transform: [{ rotate: '45deg' }]
transform: [{ scale: 1.2 }]
transform: [{ translateX: 20 }]
transform: [{ translateY: -10 }]

// دمج تحويلات متعددة
transform: [
  { rotate: '10deg' },
  { scale: 0.95 },
]

// مثال عملي — بادج مائل
badge: {
  position: 'absolute',
  top: -8,
  right: -8,
  transform: [{ rotate: '12deg' }],
}

// مثال — زر يكبر عند الضغط (مع Animated)
style={{
  transform: [{ scale: pressed ? 0.95 : 1 }],
}}

.6 overflow و zIndex overflow & zIndex

📦 overflow و zIndex
// overflow — قص المحتوى الزائد
{ overflow: 'hidden' }   // يقص كل ما يخرج عن الحاوية (الشائع)
{ overflow: 'visible' }  // يظهر المحتوى خارج الحاوية (افتراضي)
{ overflow: 'scroll' }   // نادراً — استخدم ScrollView بدلاً منه

// مثال عملي — صورة دائرية مقصوصة
avatarContainer: {
  width: 80, height: 80,
  borderRadius: 40,
  overflow: 'hidden',   // يضمن أن الصورة مقصوصة داخل الدائرة
}

// zIndex — ترتيب الطبقات (مثل z-index في CSS)
modalOverlay: { zIndex: 100 }
header: { zIndex: 50 }
content: { zIndex: 1 }

// مثال — Tooltip يظهر فوق كل شيء
tooltip: {
  position: 'absolute',
  zIndex: 999,
  backgroundColor: '#1e293b',
  padding: 8,
  borderRadius: 6,
}
overflow: 'hidden' مهم جداً عند:
• الصور الدائرية (borderRadius + overflow: 'hidden')
• بطاقات بصور في الأعلى (overflow: 'hidden' على البطاقة كلها)
• أي عنصر ترسم فيه خارج حدوده

.7 تطبيق عملي — بطاقة احترافية Pro Card

🏆 بطاقة تجمع كل الأنماط
import { LinearGradient } from 'expo-linear-gradient';

function ProCard({ title, subtitle, tag }) {
  return (
    <View style={s.card}>

      {/* صورة الغلاف — gradient */}
      <LinearGradient
        colors={['#0ea5e9', '#7c3aed']}
        style={s.cover}
      >
        <View style={s.badge}>
          <Text style={s.badgeText}>{tag}</Text>
        </View>
      </LinearGradient>

      {/* الصورة الشخصية */}
      <View style={s.avatarWrap}>
        <Image
          source={{ uri: '...' }}
          style={s.avatar}
        />
      </View>

      <View style={s.body}>
        <Text style={s.title}>{title}</Text>
        <Text style={s.sub}>{subtitle}</Text>
      </View>
    </View>
  );
}

const s = StyleSheet.create({
  card: {
    backgroundColor: 'white',
    borderRadius: 20,
    overflow: 'hidden',
    ...Platform.select({
      ios:     { shadowColor: '#000', shadowOffset: { width:0, height:8 },
                 shadowOpacity: 0.12, shadowRadius: 16 },
      android: { elevation: 6 },
    }),
  },
  cover:   { height: 100, justifyContent: 'flex-end', padding: 10 },
  badge:   { backgroundColor: 'rgba(255,255,255,.25)',
             borderRadius: 999, paddingHorizontal: 10, paddingVertical: 3,
             alignSelf: 'flex-start', transform: [{ rotate: '-2deg' }] },
  badgeText: { color: 'white', fontSize: 11, fontWeight: '700' },
  avatarWrap: { width: 64, height: 64, borderRadius: 32,
                overflow: 'hidden', borderWidth: 3, borderColor: 'white',
                marginTop: -32, marginRight: 16 },
  avatar: { width: '100%', height: '100%' },
  body:  { padding: 12 },
  title: { fontSize: 18, fontWeight: '800', color: '#0f172a' },
  sub:   { fontSize: 13, color: '#64748b', marginTop: 2 },
});

تمرين 5: تمييز بين opacity وrgba

عندما تريد خلفية زرقاء شبه شفافة للبطاقة لكن النص بداخلها واضح تماماً — أيهما تستخدم: opacity أم rgba؟ ولماذا؟

الحل:

تستخدم rgba لأن opacity تؤثر على كل الأبناء (النص سيصبح شفافاً هو الآخر)، أما rgba تُطبق الشفافية على اللون فقط دون التأثير على المحتوى.

// ✅ rgba — النص سيبقى واضحاً
{ backgroundColor: 'rgba(14, 165, 233, 0.15)' }

// ❌ opacity — النص سيصبح شفافاً أيضاً
{ opacity: 0.15 }
