.1 لماذا FlatList لا ScrollView؟ Why FlatList?

في الدروس السابقة استخدمنا ScrollView — لكنه يُحمّل كل العناصر في الذاكرة مرة واحدة. تخيّل 1000 جهة اتصال!

المعيارScrollViewFlatList ⭐
طريقة التحميلكل العناصر معاًيُحمّل ما هو مرئي فقط
الذاكرةترتفع مع الكمثابتة تقريباً
100 عنصريعمل بخيرمثالي
10,000 عنصريتجمّد / يُتعطّلسلس
Pull to Refreshيدوي معقدمدمج
Infinite Scrollيدوي معقدonEndReached مدمج
Header / Footerيدويprops مدمجة
Virtualization — FlatList يستخدم تقنية "النوافذ المتحركة": يُحمّل فقط العناصر المرئية على الشاشة + بضعة عناصر قبلها وبعدها، ثم يُعيد تدوير الخلايا القديمة للعناصر الجديدة.

.2 FlatList — الأساسيات FlatList Basics

FlatList يحتاج حداً أدنى من 3 خصائص: data، renderItem، keyExtractor:

📋 FlatList — الحد الأدنى
import { FlatList, Text, View, StyleSheet } from 'react-native';

const FRUITS = [
  { id: '1', name: 'تفاح',    color: '#f43f5e' },
  { id: '2', name: 'موز',     color: '#f59e0b' },
  { id: '3', name: 'عنب',     color: '#7c3aed' },
  { id: '4', name: 'برتقال', color: '#ea580c' },
];

function FruitItem({ item }) {
  return (
    <View style={[s.item, { borderRightColor: item.color }]}>
      <Text style={s.name}>{item.name}</Text>
    </View>
  );
}

export function FruitList() {
  return (
    <FlatList
      data={FRUITS}                     // ✅ مصفوفة البيانات
      renderItem={({ item }) => (       // ✅ كيف يُعرض كل عنصر
        <FruitItem item={item} />
      )}
      keyExtractor={item => item.id}    // ✅ مفتاح فريد لكل عنصر
    />
  );
}

const s = StyleSheet.create({
  item: {
    padding: 16, backgroundColor: 'white',
    borderRightWidth: 4, marginBottom: 8, borderRadius: 10,
  },
  name: { fontSize: 16, fontWeight: '700' },
});
keyExtractor — يُعطي FlatList مفتاحاً فريداً لكل عنصر حتى يعرف أيها تغيّر عند التحديث. دائماً استخدم item.id.toString() أو أي حقل فريد. إذا كانت البيانات بدون id استخدم الـ index: (item, index) => index.toString() لكن هذا أقل كفاءة.

.4 Pull to Refresh و Infinite Scroll Dynamic Loading

🔄 Pull to Refresh — سحب للتحديث
function NewsScreen() {
  const [news, setNews] = useState([]);
  const [refreshing, setRefreshing] = useState(false);

  const fetchNews = async () => {
    const data = await api.getNews();
    setNews(data);
  };

  const handleRefresh = async () => {
    setRefreshing(true);    // يُظهر المؤشر الدوار
    await fetchNews();
    setRefreshing(false);   // يُخفي المؤشر
  };

  return (
    <FlatList
      data={news}
      renderItem={({ item }) => <NewsCard item={item} />}
      keyExtractor={item => item.id}
      refreshing={refreshing}         // ✅ حالة التحديث
      onRefresh={handleRefresh}       // ✅ دالة التحديث
    />
  );
}
♾️ Infinite Scroll — تحميل لا نهائي
function ProductsScreen() {
  const [products, setProducts] = useState([]);
  const [page, setPage]         = useState(1);
  const [loading, setLoading]   = useState(false);
  const [hasMore, setHasMore]   = useState(true);

  const loadMore = async () => {
    if (loading || !hasMore) return;  // لا تُكرّر الطلب
    setLoading(true);

    const newData = await api.getProducts({ page });
    if (newData.length === 0) {
      setHasMore(false);  // وصلنا لآخر البيانات
    } else {
      setProducts(prev => [...prev, ...newData]);  // أضف للقائمة
      setPage(p => p + 1);
    }
    setLoading(false);
  };

  return (
    <FlatList
      data={products}
      renderItem={({ item }) => <ProductCard item={item} />}
      keyExtractor={item => item.id}
      onEndReached={loadMore}         // ✅ يُطلق عند الاقتراب من النهاية
      onEndReachedThreshold={0.5}    // ✅ 50% من الأسفل
      ListFooterComponent={           // مؤشر التحميل في الأسفل
        loading
          ? <ActivityIndicator size="large" color="#0ea5e9" style={{ py: 20 }} />
          : !hasMore && <Text style={s.endMsg}>وصلت للنهاية ✓</Text>
      }
    />
  );
}

🌐 جلب بيانات من API — النمط الكامل
import { useState, useEffect } from 'react';
import { FlatList, View, Text, Image, ActivityIndicator, StyleSheet } from 'react-native';

function UsersScreen() {
  const [users, setUsers]     = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError]     = useState(null);

  useEffect(() => {
    fetchUsers();
  }, []);

  const fetchUsers = async () => {
    try {
      setLoading(true);
      setError(null);

      const res  = await fetch('https://jsonplaceholder.typicode.com/users');
      const data = await res.json();
      setUsers(data);
    } catch (err) {
      setError('فشل تحميل البيانات — تحقق من الاتصال');
    } finally {
      setLoading(false);
    }
  };

  // حالة التحميل
  if (loading) return (
    <View style={s.center}>
      <ActivityIndicator size="large" color="#0ea5e9" />
    </View>
  );

  // حالة الخطأ
  if (error) return (
    <View style={s.center}>
      <Text style={s.errorText}>{error}</Text>
      <TouchableOpacity onPress={fetchUsers} style={s.retryBtn}>
        <Text style={{ color: 'white' }}>إعادة المحاولة</Text>
      </TouchableOpacity>
    </View>
  );

  // البيانات جاهزة
  return (
    <FlatList
      data={users}
      keyExtractor={item => item.id.toString()}
      renderItem={({ item }) => (
        <View style={s.userCard}>
          <View style={s.avatar}>
            <Text style={s.avatarText}>{item.name[0]}</Text>
          </View>
          <View>
            <Text style={s.name}>{item.name}</Text>
            <Text style={s.email}>{item.email}</Text>
          </View>
        </View>
      )}
      ItemSeparatorComponent={() => <View style={s.sep} />}
    />
  );
}
لا تنسَ حالات التحميل والخطأ! تطبيق بدون loading state يظهر شاشة فارغة مربكة. دائماً: loading → مؤشر دوار، error → رسالة + زر إعادة، empty → رسالة توجيهية.

.6 SectionList — قوائم المجموعات SectionList

SectionList مثل FlatList لكن للقوائم المقسّمة إلى مجموعات — قائمة جهات الاتصال بالحروف، قائمة منتجات بالفئات، جدول أعمال بالتواريخ:

function groupByLetter(contacts) {

📱 ContactsApp — التطبيق الكامل
import { useState, useMemo, useCallback } from 'react';
import {
  SectionList, TextInput, View, Text,
  TouchableOpacity, StyleSheet,
} from 'react-native';

const COLORS = ['#0ea5e9','#7c3aed','#16a34a','#ea580c','#f43f5e'];
const getColor = name => COLORS[name.charCodeAt(0) % COLORS.length];

export function ContactsApp({ contacts }) {
  const [search, setSearch] = useState('');

  // فلترة وتجميع — يُعاد احتسابه فقط عند تغيير البيانات
  const sections = useMemo(() => {
    const filtered = contacts.filter(c =>
      c.name.includes(search) || c.phone.includes(search)
    );
    const grouped = filtered.reduce((acc, c) => {
      const l = c.name[0];
      if (!acc[l]) acc[l] = [];
      acc[l].push(c);
      return acc;
    }, {});
    return Object.keys(grouped).sort().map(l => ({ title: l, data: grouped[l] }));
  }, [contacts, search]);

  // useCallback يمنع إعادة إنشاء الدالة مع كل render
  const renderItem = useCallback(({ item }) => (
    <TouchableOpacity
      style={s.item}
      onPress={() => console.log('اتصل بـ', item.name)}
    >
      <View style={[s.avatar, { backgroundColor: getColor(item.name) }]}>
        <Text style={s.avatarText}>{item.name[0]}</Text>
      </View>
      <View style={{ flex: 1 }}>
        <Text style={s.name}>{item.name}</Text>
        <Text style={s.phone}>{item.phone}</Text>
      </View>
      <Text style={s.callIcon}>📞</Text>
    </TouchableOpacity>
  ), []);

  const renderHeader = useCallback(({ section: { title, data } }) => (
    <View style={s.sectionHeader}>
      <Text style={s.sectionTitle}>{title}</Text>
      <Text style={s.sectionCount}>{data.length}</Text>
    </View>
  ), []);

  return (
    <View style={{ flex: 1 }}>
      <TextInput
        style={s.searchInput}
        placeholder="🔍 ابحث في جهات الاتصال..."
        value={search}
        onChangeText={setSearch}
        placeholderTextColor="#94a3b8"
      />

      <SectionList
        sections={sections}
        keyExtractor={item => item.id}
        renderItem={renderItem}
        renderSectionHeader={renderHeader}
        stickySectionHeadersEnabled
        ItemSeparatorComponent={() => <View style={s.sep} />}
        ListEmptyComponent={
          <View style={s.empty}>
            <Text style={s.emptyText}>لا نتائج لـ "{search}"</Text>
          </View>
        }
      />
    </View>
  );
}

const s = StyleSheet.create({
  searchInput: {
    margin: 12, padding: 12, borderRadius: 12,
    borderWidth: 2, borderColor: '#e2e8f0', backgroundColor: 'white',

  },

تمرين 5: Infinite Scroll

اكتب FlatList مع Infinite Scroll يُحاكي تحميل بيانات: صفحة 1 تُحمّل 10 عناصر، وعند الوصول للنهاية يُحمّل 10 أخرى، وهكذا حتى 50 عنصراً إجمالاً.

الحل: 
const PAGE_SIZE = 10;
const MAX_ITEMS = 50;

function generateItems(page) {
  return Array.from({ length: PAGE_SIZE }, (_, i) => ({
    id: `${page}-${i}`,
    text: `عنصر ${(page - 1) * PAGE_SIZE + i + 1}`,
  }));
}

function InfiniteList() {
  const [items, setItems]   = useState(() => generateItems(1));
  const [page, setPage]     = useState(2);
  const [loading, setLoading] = useState(false);

  const hasMore = items.length < MAX_ITEMS;

  const loadMore = async () => {
    if (loading || !hasMore) return;
    setLoading(true);
    // محاكاة تأخر الشبكة
    await new Promise(r => setTimeout(r, 1000));
    setItems(prev => [...prev, ...generateItems(page)]);
    setPage(p => p + 1);
    setLoading(false);
  };

  return (
    <FlatList
      data={items}
      keyExtractor={item => item.id}
      renderItem={({ item }) => (
        <View style={{ padding: 16, borderBottomWidth: 1, borderBottomColor: '#e2e8f0' }}>
          <Text>{item.text}</Text>
        </View>
      )}
      onEndReached={loadMore}
      onEndReachedThreshold={0.3}
      ListFooterComponent={
        loading ? <ActivityIndicator color="#0ea5e9" style={{ padding: 20 }} />
        : !hasMore ? <Text style={{ textAlign: 'center', padding: 20, color: '#94a3b8' }}>انتهت القائمة ✓</Text>
        : null
      }
    />
  );
}

اختبر فهمك لـ FlatList و SectionList!

1ScrollView أكفأ من FlatList عند عرض 1000 عنصر لأنه يُحمّل كل شيء مرة واحدة.
2SectionList تأخذ خاصية sections بدل data.
3onEndReached تُطلق عند اقتراب المستخدم من نهاية القائمة.
4يمكن استخدام numColumns مع horizontal في نفس الوقت.
5استخدام useCallback في renderItem يُحسّن أداء FlatList.
